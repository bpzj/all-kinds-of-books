\newcommand\test[3]{{
  \foreach \x in {#1, #2, #3}{
    \settowidth{\textlen}{\x}
    \ifdim \textlen > \lablen
      \setlength{\lablen}{\textlen}
    \else
      \relax
    \fi
  }
  \makebox[\the\lablen][s]{#1}\\
  \makebox[\the\lablen][s]{#2}\\
  \makebox[\the\lablen][s]{#3}\\
}}
